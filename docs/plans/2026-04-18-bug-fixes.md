## Bug 1 — Draft stories visible to reviewers

**Symptom:** Reviewers see stories the reporter has not yet submitted (status = `draft`).

**Root cause:** `admin_list_stories` has two paths. The "normal" path uses `_build_story_query()` which excludes drafts via the default `exclude_drafts=True`. The "delta mode" path (when `updated_since` is passed by the SWR cache) bypasses that helper and queries `Story.organization_id == org_id` only — drafts leak through.

**Files:**
- Modify: `api/app/routers/admin/stories.py:61-68` (delta query)
- Test: `api/tests/test_admin_stories_delta.py` (new)

**Step 1: Write the failing test**

```python
# Odia Unicode range: U+0B00–U+0B7F
def _is_predominantly_odia(text: str, threshold: float = 0.4) -> bool:
    """Return True if at least `threshold` fraction of letters are Odia script."""
    if not text:
        return False
    letters = [c for c in text if c.isalpha()]
    if not letters:
        return False
    odia = sum(1 for c in letters if "\u0B00" <= c <= "\u0B7F")
    return (odia / len(letters)) >= threshold
```

```javascript
// Bump default; long story translations need ≥6k tokens
const DEFAULT_MAX_TOKENS = 8192;

export async function llmChat({ messages, max_tokens, temperature, expectOdia = false }) {
  const body = {
    messages,
    max_tokens: max_tokens ?? DEFAULT_MAX_TOKENS,
    temperature: temperature ?? 0.3,
  };
  let response = await apiPost('/sarvam/chat', body);
  let text = response?.choices?.[0]?.message?.content ?? '';
  const finishReason = response?.choices?.[0]?.finish_reason;

  // Retry once on truncation or (if expected) wrong language
  const truncated = finishReason === 'length';
  const wrongLang = expectOdia && !_isPredominantlyOdia(text);
  if (truncated || wrongLang) {
    const retryBody = { ...body, max_tokens: Math.min((body.max_tokens ?? 0) * 2, 16384) };
    if (wrongLang) {
      // Strengthen the system prompt for the retry
      retryBody.messages = [
        ...messages.filter((m) => m.role !== 'system'),
        { role: 'system', content: 'You MUST respond in Odia (ଓଡ଼ିଆ) script only. Do not use English or transliteration.' },
      ];
    }
    response = await apiPost('/sarvam/chat', retryBody);
    text = response?.choices?.[0]?.message?.content ?? '';
  }
  return text;
}

function _isPredominantlyOdia(text, threshold = 0.4) {
  if (!text) return false;
  const letters = [...text].filter((c) => /\p{L}/u.test(c));
  if (letters.length === 0) return false;
  const odia = letters.filter((c) => c >= '\u0B00' && c <= '\u0B7F').length;
  return odia / letters.length >= threshold;
}
```
